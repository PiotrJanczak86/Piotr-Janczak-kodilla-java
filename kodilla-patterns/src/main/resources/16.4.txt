SELECT I.SUMMARY, L.NAME
FROM ISSUES I
JOIN ISSUESLISTS L ON L.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, U.ID
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NO_OF_TASKS
FROM USERS U JOIN ISSUES ON U.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY ISSUES.USER_ID_ASSIGNEDTO;
